00068   def user_has_access(self, user):
00069     """Returns true if the given user has access to this task list."""
00070     if not user: return False
00071     query = db.Query(TaskListMember)
00072     query.filter('task_list =', self)
00073     query.filter('user =', user)

00145 class TaskListPage(BaseRequestHandler):
00146   """Displays a single task list based on ID.
00147 
00148   If the task list is not published, we give a 403 unless the user is a
00149   collaborator on the list. If it is published, but the user is not a
00150   collaborator, we show the more limited HTML view of the task list rather
00151   than the interactive AJAXy edit page.
00152   """
00153 
00154   # The different task list output types we support: content types and
00155   # template file extensions
00156   _OUTPUT_TYPES = {
00157     'default': ['text/html', 'html'],
00158     'html': ['text/html', 'html'],
00159     'atom': ['application/atom+xml', 'xml'],
00160   }
00161 
00162   def get(self):
00163     task_list = TaskList.get(self.request.get('id'))
00164     if not task_list:
00165       self.error(403)
00166       return
00167 
00168     # Choose a template based on the output type
00169     output_name = self.request.get('output')
00170     output_name_list = TaskListPage._OUTPUT_TYPES.keys()
00171     if output_name not in output_name_list:
00172       output_name = output_name_list[0]
00173     output_type = TaskListPage._OUTPUT_TYPES[output_name]
00174 
00175     # Validate this user has access to this task list. If not, they can
00176     # access the html view of this list only if it is published.
00177     if not task_list.current_user_has_access():
00178       if task_list.published:
00179         if output_name == 'default':
00180           output_name = 'html'
00181           output_type = TaskListPage._OUTPUT_TYPES[output_name]
00182       else:
00183         user = users.GetCurrentUser()
00184         if not user:
00185           self.redirect(users.CreateLoginURL(self.request.uri))
00186         else:
00187           self.error(403)
00188         return
00189 
00190     # Filter out archived tasks by default
00191     show_archive = self.request.get('archive')
00192     tasks = task_list.task_set.order('-priority').order('created')
00193     if not show_archive:
00194       tasks.filter('archived =', False)
00195     tasks = list(tasks)
00196 
00197     # Get the last updated date from the list of tasks
00198     if len(tasks) > 0:
00199       updated = max([task.updated for task in tasks])
00200     else:
00201       updated = None
00202 
00203     self.response.headers['Content-Type'] = output_type[0]
00204     self.generate('tasklist_' + output_name + '.' + output_type[1], {
00205       'task_list': task_list,
00206       'tasks': tasks,
00207       'archive': show_archive,
00208       'updated': updated,
00209     })

00307 class InboxAction(BaseRequestHandler):
00308   """Performs an action in the user's TaskList inbox.
00309 
00310   We support Archive, Unarchive, and Delete actions. The action is specified
00311   by the "action" argument in the POST. The names are capitalized because
00312   they correspond to the text in the buttons in the form, which all have the
00313   name "action".
00314   """
00315   def post(self):
00316     action = self.request.get('action')
00317     lists = self.request.get('list', allow_multiple=True)
00318     if not action in ['Archive', 'Unarchive', 'Delete']:
00319       self.error(403)
00320       return
00321 
00322     for key in lists:
00323       task_list = TaskList.get(key)
00324 
00325       # Validate this user has access to this task list
00326       if not task_list or not task_list.current_user_has_access():
00327         self.error(403)
00328         return
00329 
00330       if action == 'Archive':
00331         task_list.archived = True
00332         task_list.put()
00333       elif action == 'Unarchive':
00334         task_list.archived = False
00335         task_list.put()
00336       else:
00337         for member in task_list.tasklistmember_set:
00338           member.delete()
00339         for task in task_list.task_set:
00340           task.delete()
00341         task_list.delete()
00342 
00343     self.redirect(self.request.get('next'))
