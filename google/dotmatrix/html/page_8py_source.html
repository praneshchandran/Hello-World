00031 def ForceLogin(func):
00032   """Decorator for get/post: If the user isn't logged in, take them to the
00033   login page.  Otherwise do normal processing.
00034   """
00035   def Wrapped(self, *arg, **kwarg):
00036     if users.GetCurrentUser() is None:
00037       self.redirect(users.CreateLoginURL(self.request.uri))
00038     else:
00039       # If the user came back to this page after logout, they'd get an error,
00040       # so send them somewhere else.
00041       self.after_logout_url = '/list'
00042       return func(self, *arg, **kwarg)
00043   return Wrapped

00074   def RenderError(self, error_message, status_code=None):
00075     """Render a stark error message."""
00076     if status_code is not None:
00077       self.error(status_code)
00078     values = {'error_message': error_message}
00079     self.RenderTemplate('error.html', values)

00081   def RenderTemplate(self, template_filename, template_values):
00082     """Render a template.  Takes care of some of the boilerplate (like getting
00083     the user, login links, showing error messages, etc.) for you."""
00084     path = os.path.join(self.TEMPLATE_PATH, template_filename)
00085     if not self.after_logout_url:
00086       self.after_logout_url = self.request.uri
00087     common_values = dict(
00088         user=users.GetCurrentUser(),
00089         user_is_admin=users.IsCurrentUserAdmin(),
00090         logout_url=users.CreateLogoutURL(self.after_logout_url),
00091         login_url=users.CreateLoginURL(self.request.uri),
00092         errors=self.errors,
00093         elapsed_time=clock.Elapsed()
00094         )
00095     common_values.update(template_values)
00096     self.response.out.write(template.render(path, common_values))
