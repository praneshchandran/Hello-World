00089 def GetListContext(request):
00090   """Get the template values expected by _list.html."""
00091   number = int(request.get('n', 30))
00092   offset = int(request.get('offset', 0))
00093   max_index = models.DateSortIndex.current()
00094   threshold = max_index - offset
00095   # BUG: This crashes on the last page.  Falling off the list somehow?
00096   pics = models.Picture.gql('WHERE date_sort_index < :1 '
00097                             'ORDER BY date_sort_index DESC', threshold)
00098   pics = pics.fetch(number +1)
00099   show_next = False
00100   if len(pics) > number:
00101     show_next = True
00102     pics.pop()
00103   context = {
00104     'pics': pics,
00105     'show_next': show_next,
00106     'show_prev': (offset > 0),
00107     'next_offset': number + offset,
00108     'prev_offset': max(0, offset - number),
00109   }
00110   return context

00128   def get(self):
00129     path = self.request.path.split('/')
00130     name = path[-1]
00131     # Note: have to call unquote twice to get back to non-quoted name.
00132     name = urllib.unquote(urllib.unquote(name))
00133     name = name.replace('.png', '')
00134     pics = models.Picture.GetByName(name)
00135     if pics.count() < 1:
00136       self.RenderError('Image not found.', status_code=404)
00137       return
00138     pic = pics[0]
00139     self.response.headers['content-disposition'] = 'inline'
00140     self.response.headers['filename'] = '%s.png' % name
00141     self.response.headers['content-type'] = 'image/png'
00142     self.SetCache(604800)  # 1 week
00143     self.response.headers['last-modified'] = self.FormatDate(pic.date)
00144     self.response.out.write(pic.png_data)

00150     def ColorToggle():
00151       """Give chessboard alternating colors."""
00152       while True:
00153         for i in range(4):  # First row (starts with white)
00154           yield 'White'
00155           yield 'Black'
00156         for i in range(4):  # Second row (starts with black)
00157           yield 'Black'
00158           yield 'White'
00159 
00160     row = ['Rook', 'Knight', 'Bishop', 'Queen',
00161            'King', 'Bishop', 'Knight', 'Rook']
00162     color = ColorToggle()
00163     chessboard = [
00164       ['/%s Black on %s.png' % (piece, color.next()) for piece in row],
00165       ['/Pawn Black on %s.png' % color.next() for x in range(8)],
00166       ['/Empty %s.png' % color.next() for x in range(8)],
00167       ['/Empty %s.png' % color.next() for x in range(8)],
00168       ['/Empty %s.png' % color.next() for x in range(8)],
00169       ['/Empty %s.png' % color.next() for x in range(8)],
00170       ['/Pawn White on %s.png' % color.next() for x in range(8)],
00171       ['/%s White on %s.png' % (piece, color.next()) for piece in row],
00172     ]
00173     return chessboard
