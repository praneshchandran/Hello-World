00091 class Session(db.Model):
00092   """A shell session. Stores the session's globals.
00093 
00094   Each session globals is stored in one of two places:
00095 
00096   If the global is picklable, it's stored in the parallel globals and
00097   global_names list properties. (They're parallel lists to work around the
00098   unfortunate fact that the datastore can't store dictionaries natively.)
00099 
00100   If the global is not picklable (e.g. modules, classes, and functions), or if
00101   it was created by the same statement that created an unpicklable global,
00102   it's not stored directly. Instead, the statement is stored in the
00103   unpicklables list property. On each request, before executing the current
00104   statement, the unpicklable statements are evaluated to recreate the
00105   unpicklable globals.
00106 
00107   The unpicklable_names property stores all of the names of globals that were
00108   added by unpicklable statements. When we pickle and store the globals after
00109   executing a statement, we skip the ones in unpicklable_names.
00110 
00111   Using Text instead of string is an optimization. We don't query on any of
00112   these properties, so they don't need to be indexed.
00113   """
00114   global_names = db.ListProperty(db.Text)
00115   globals = db.ListProperty(db.Blob)
00116   unpicklable_names = db.ListProperty(db.Text)
00117   unpicklables = db.ListProperty(db.Text)
00118 
00119   def set_global(self, name, value):
00120     """Adds a global, or updates it if it already exists.
00121 
00122     Also removes the global from the list of unpicklable names.
00123 
00124     Args:
00125       name: the name of the global to remove
00126       value: any picklable value
00127     """
00128     blob = db.Blob(pickle.dumps(value))
00129 
00130     if name in self.global_names:
00131       index = self.global_names.index(name)
00132       self.globals[index] = blob
00133     else:
00134       self.global_names.append(db.Text(name))
00135       self.globals.append(blob)
00136 
00137     self.remove_unpicklable_name(name)
00138 
00139   def remove_global(self, name):
00140     """Removes a global, if it exists.
00141 
00142     Args:
00143       name: string, the name of the global to remove
00144     """
00145     if name in self.global_names:
00146       index = self.global_names.index(name)
00147       del self.global_names[index]
00148       del self.globals[index]
00149 
00150   def globals_dict(self):
00151     """Returns a dictionary view of the globals.
00152     """
00153     return dict((name, pickle.loads(val))
00154                 for name, val in zip(self.global_names, self.globals))
00155 
00156   def add_unpicklable(self, statement, names):
00157     """Adds a statement and list of names to the unpicklables.
00158 
00159     Also removes the names from the globals.
00160 
00161     Args:
00162       statement: string, the statement that created new unpicklable global(s).
00163       names: list of strings; the names of the globals created by the statement.
00164     """
00165     self.unpicklables.append(db.Text(statement))
00166 
00167     for name in names:
00168       self.remove_global(name)
00169       if name not in self.unpicklable_names:
00170         self.unpicklable_names.append(db.Text(name))
00171 
00172   def remove_unpicklable_name(self, name):
00173     """Removes a name from the list of unpicklable names, if it exists.
00174 
00175     Args:
00176       name: string, the name of the unpicklable global to remove
00177     """
00178     if name in self.unpicklable_names:
00179       self.unpicklable_names.remove(name)

00211 class StatementHandler(webapp.RequestHandler):
00212   """Evaluates a python statement in a given session and returns the result.
00213   """
00214 
00215   def get(self):
00216     self.response.headers['Content-Type'] = 'text/plain'
00217 
00218     # extract the statement to be run
00219     statement = self.request.get('statement')
00220     if not statement:
00221       return
00222 
00223     # the python compiler doesn't like network line endings
00224     statement = statement.replace('\r\n', '\n')
00225 
00226     # add a couple newlines at the end of the statement. this makes
00227     # single-line expressions such as 'class Foo: pass' evaluate happily.
00228     statement += '\n\n'
00229 
00230     # log and compile the statement up front
00231     try:
00232       logging.info('Compiling and evaluating:\n%s' % statement)
00233       compiled = compile(statement, '<string>', 'single')
00234     except:
00235       self.response.out.write(traceback.format_exc())
00236       return
00237 
00238     # create a dedicated module to be used as this statement's __main__
00239     statement_module = new.module('__main__')
00240 
00241     # use this request's __builtin__, since it changes on each request.
00242     # this is needed for import statements, among other things.
00243     import __builtin__
00244     statement_module.__builtins__ = __builtin__
00245 
00246     # load the session from the datastore
00247     session = Session.get(self.request.get('session'))
00248 
00249     # swap in our custom module for __main__. then unpickle the session
00250     # globals, run the statement, and re-pickle the session globals, all
00251     # inside it.
00252     old_main = sys.modules.get('__main__')
00253     try:
00254       sys.modules['__main__'] = statement_module
00255       statement_module.__name__ = '__main__'
00256 
00257       # re-evaluate the unpicklables
00258       for code in session.unpicklables:
00259         exec code in statement_module.__dict__
00260 
00261       # re-initialize the globals
00262       for name, val in session.globals_dict().items():
00263         try:
00264           statement_module.__dict__[name] = val
00265         except:
00266           msg = 'Dropping %s since it could not be unpickled.\n' % name
00267           self.response.out.write(msg)
00268           logging.warning(msg + traceback.format_exc())
00269           session.remove_global(name)
00270 
00271       # run!
00272       old_globals = dict(statement_module.__dict__)
00273       try:
00274         old_stdout = sys.stdout
00275         old_stderr = sys.stderr
00276         try:
00277           sys.stdout = self.response.out
00278           sys.stderr = self.response.out
00279           exec compiled in statement_module.__dict__
00280         finally:
00281           sys.stdout = old_stdout
00282           sys.stderr = old_stderr
00283       except:
00284         self.response.out.write(traceback.format_exc())
00285         return
00286 
00287       # extract the new globals that this statement added
00288       new_globals = {}
00289       for name, val in statement_module.__dict__.items():
00290         if name not in old_globals or val != old_globals[name]:
00291           new_globals[name] = val
00292 
00293       if True in [isinstance(val, UNPICKLABLE_TYPES)
00294                   for val in new_globals.values()]:
00295         # this statement added an unpicklable global. store the statement and
00296         # the names of all of the globals it added in the unpicklables.
00297         session.add_unpicklable(statement, new_globals.keys())
00298         logging.debug('Storing this statement as an unpicklable.')
00299 
00300       else:
00301         # this statement didn't add any unpicklables. pickle and store the
00302         # new globals back into the datastore.
00303         for name, val in new_globals.items():
00304           if not name.startswith('__'):
00305             session.set_global(name, val)
00306 
00307     finally:
00308       sys.modules['__main__'] = old_main
00309 
00310     session.put()
