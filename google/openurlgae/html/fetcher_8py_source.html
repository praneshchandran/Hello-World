00032 class UrlfetchFetcher(fetchers.HTTPFetcher):
00033   """An HTTPFetcher subclass that uses Google App Engine's urlfetch module.
00034   """
00035   def fetch(self, url, body=None, headers=None):
00036     """
00037     This performs an HTTP POST or GET, following redirects along
00038     the way. If a body is specified, then the request will be a
00039     POST. Otherwise, it will be a GET.
00040 
00041     @param headers: HTTP headers to include with the request
00042     @type headers: {str:str}
00043 
00044     @return: An object representing the server's HTTP response. If
00045       there are network or protocol errors, an exception will be
00046       raised. HTTP error responses, like 404 or 500, do not
00047       cause exceptions.
00048 
00049     @rtype: L{HTTPResponse}
00050 
00051     @raise Exception: Different implementations will raise
00052       different errors based on the underlying HTTP library.
00053     """
00054     if not fetchers._allowedURL(url):
00055       raise ValueError('Bad URL scheme: %r' % (url,))
00056 
00057     if body:
00058       method = urlfetch.POST
00059       if 'Content-Type' not in headers:
00060         headers['Content-Type'] = 'application/x-www-form-urlencoded'
00061     else:
00062       method = urlfetch.GET
00063 
00064     if not headers:
00065       headers = {}
00066 
00067     # follow up to 10 redirects
00068     for i in range(10):
00069       resp = urlfetch.fetch(url, body, method, headers)
00070       if resp.status_code in (301, 302):
00071         logging.debug('Following %d redirect to %s' %
00072                       (resp.status_code, resp.headers['location']))
00073         url = resp.headers['location']
00074       else:
00075         break
00076 
00077     return fetchers.HTTPResponse(url, resp.status_code, resp.headers,
00078                                  resp.content)
