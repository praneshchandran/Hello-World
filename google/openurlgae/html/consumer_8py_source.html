00188   def relative_time(self, timestamp):
00189     """Returns a friendly string describing how long ago the timestamp was.
00190 
00191     Args:
00192       timestamp: a datetime
00193 
00194     Returns:
00195       string
00196     """
00197     def format_number(num):
00198       if num <= 9:
00199         return {1: 'one', 2: 'two', 3: 'three', 4: 'four', 5: 'five',
00200                 6: 'six', 7: 'seven', 8: 'eight', 9: 'nine'}[num]
00201       else:
00202         return str(num)
00203 
00204     delta = datetime.datetime.now() - timestamp
00205     minutes = delta.seconds / 60
00206     hours = minutes / 60
00207 
00208     if delta.days > 1:
00209       return '%s days ago' % format_number(delta.days)
00210     elif delta.days == 1:
00211       return 'yesterday'
00212     elif hours > 1:
00213       return '%s hours ago' % format_number(hours)
00214     elif hours == 1:
00215       return 'an hour ago'
00216     elif minutes > 25:
00217       return 'half an hour ago'
00218     elif minutes > 5:
00219       return '%s minutes ago' % format_number(minutes)
00220     else:
00221       return 'moments ago'

00223   def display_name(self, openid_url):
00224     """Extracts a short, representative part of an OpenID URL for display.
00225 
00226     For example, it returns "ryan" for:
00227       ryan.com
00228       www.ryan.com
00229       ryan.provider.com
00230       provider.com/ryan
00231       provider.com/id/path/ryan
00232 
00233     Adapted from Net::OpenID::Consumer, by Brad Fitzpatrick. See:
00234 
00235     http://code.sixapart.com/svn/openid/trunk/perl/Net-OpenID-Consumer/lib/Net/OpenID/VerifiedIdentity.pm
00236 
00237     Args:
00238       openid_url: string
00239 
00240     Returns:
00241       string
00242     """
00243     if not openid_url:
00244       return 'None'
00245 
00246     username_re = '[\w.+-]+'
00247 
00248     scheme, host, path, params, query, frag = urlparse.urlparse(openid_url)
00249 
00250     def sanitize(display_name):
00251       if '@' in display_name:
00252         # don't display full email addresses; use just the user name part
00253         display_name = display_name[:display_name.index('@')]
00254       return display_name
00255 
00256     # is the username in the params?
00257     match = re.search('(u|id|user|userid|user_id|profile)=(%s)' % username_re,
00258                       path)
00259     if match:
00260       return sanitize(match.group(2))
00261 
00262     # is the username in the path?
00263     path = path.split('/')
00264     if re.match(username_re, path[-1]):
00265       return sanitize(path[-1])
00266 
00267     # use the hostname
00268     host = host.split('.')
00269     if len(host) == 1:
00270       return host[0]
00271 
00272     # strip common tlds and country code tlds
00273     common_tlds = ('com', 'org', 'net', 'edu', 'info', 'biz', 'gov', 'mil')
00274     if host[-1] in common_tlds or len(host[-1]) == 2:
00275       host = host[:-1]
00276     if host[-1] == 'co':
00277       host = host[:-1]
00278 
00279     # strip www prefix
00280     if host[0] == 'www':
00281       host = host[1:]
00282 
00283     return sanitize('.'.join(host))

00295   def post(self):
00296     """Handles login requests."""
00297     logging.info(self.args_to_dict())
00298     openid_url = self.request.get('openid')
00299     if not openid_url:
00300       self.report_error('Please enter an OpenID URL.')
00301       return
00302 
00303     logging.debug('Beginning discovery for OpenID %s' % openid_url)
00304     try:
00305       consumer = self.get_consumer()
00306       if not consumer:
00307         return
00308       auth_request = consumer.begin(openid_url)
00309     except discover.DiscoveryFailure, e:
00310       self.report_error('Error during OpenID provider discovery.', e)
00311       return
00312     except discover.XRDSError, e:
00313       self.report_error('Error parsing XRDS from provider.', e)
00314       return
00315 
00316     self.session.claimed_id = auth_request.endpoint.claimed_id
00317     self.session.server_url = auth_request.endpoint.server_url
00318     self.session.store_and_display = self.request.get('display', 'no') != 'no'
00319     self.store_session()
00320 
00321     sreg_request = sreg.SRegRequest(optional=['nickname', 'fullname', 'email'])
00322     auth_request.addExtension(sreg_request)
00323 
00324     pape_request = pape.Request([pape.AUTH_MULTI_FACTOR,
00325                                  pape.AUTH_MULTI_FACTOR_PHYSICAL,
00326                                  pape.AUTH_PHISHING_RESISTANT,
00327                                  ])
00328     auth_request.addExtension(pape_request)
00329 
00330     parts = list(urlparse.urlparse(self.request.uri))
00331     parts[2] = 'finish'
00332     parts[4] = 'session_id=%d' % self.session.key().id()
00333     parts[5] = ''
00334     return_to = urlparse.urlunparse(parts)
00335     realm = urlparse.urlunparse(parts[0:2] + [''] * 4)
00336 
00337     redirect_url = auth_request.redirectURL(realm, return_to)
00338     logging.debug('Redirecting to %s' % redirect_url)
00339     self.response.set_status(302)
00340     self.response.headers['Location'] = redirect_url
