00063   def storeAssociation(self, server_url, association):
00064     """
00065     This method puts a C{L{Association <openid.association.Association>}}
00066     object into storage, retrievable by server URL and handle.
00067     """
00068     assoc = Association(url=server_url,
00069                         handle=association.handle,
00070                         association=association.serialize())
00071     assoc.put()

00073   def getAssociation(self, server_url, handle=None):
00074     """
00075     This method returns an C{L{Association <openid.association.Association>}}
00076     object from storage that matches the server URL and, if specified, handle.
00077     It returns C{None} if no such association is found or if the matching
00078     association is expired.
00079 
00080     If no handle is specified, the store may return any association which
00081     matches the server URL. If multiple associations are valid, the
00082     recommended return value for this method is the one that will remain valid
00083     for the longest duration.
00084     """
00085     query = Association.all().filter('url', server_url)
00086     if handle:
00087       query.filter('handle', handle)
00088 
00089     results = query.fetch(1)
00090     if results:
00091       association = OpenIDAssociation.deserialize(results[0].association)
00092       if association.getExpiresIn() > 0:
00093         # hasn't expired yet
00094         return association
00095 
00096     return None

00107   def useNonce(self, server_url, timestamp, salt):
00108     """Called when using a nonce.
00109 
00110     This method should return C{True} if the nonce has not been
00111     used before, and store it for a while to make sure nobody
00112     tries to use the same value again.  If the nonce has already
00113     been used or the timestamp is not current, return C{False}.
00114 
00115     You may use L{openid.store.nonce.SKEW} for your timestamp window.
00116 
00117     @change: In earlier versions, round-trip nonces were used and
00118        a nonce was only valid if it had been previously stored
00119        with C{storeNonce}.  Version 2.0 uses one-way nonces,
00120        requiring a different implementation here that does not
00121        depend on a C{storeNonce} call.  (C{storeNonce} is no
00122        longer part of the interface.)
00123 
00124     @param server_url: The URL of the server from which the nonce
00125         originated.
00126 
00127     @type server_url: C{str}
00128 
00129     @param timestamp: The time that the nonce was created (to the
00130         nearest second), in seconds since January 1 1970 UTC.
00131     @type timestamp: C{int}
00132 
00133     @param salt: A random string that makes two nonces from the
00134         same server issued during the same second unique.
00135     @type salt: str
00136 
00137     @return: Whether or not the nonce was valid.
00138 
00139     @rtype: C{bool}
00140     """
00141     query = UsedNonce.gql(
00142       'WHERE server_url = :1 AND salt = :2 AND timestamp >= :3',
00143       server_url, salt, self._expiration_datetime())
00144     return query.fetch(1) == []

00201   def _cleanup_batch(self, query):
00202     """Deletes the first batch of entities that match the given query.
00203 
00204     Returns the number of entities that were deleted.
00205     """
00206     to_delete = list(query.fetch(CLEANUP_BATCH_SIZE))
00207 
00208     # can't use batch delete since they're all root entities :/
00209     for entity in to_delete:
00210       entity.delete()
00211 
00212     return len(to_delete)
