00056                                display_user.email() not in user_list):
00057       display_user = user_list.values()[0]
00058     template_values = {
00059       'users': user_list.values(),

00062       'login_url': users.CreateLoginURL(self.request.uri),

00070 class RPCHandler(webapp.RequestHandler):
00071   def get(self):
00072     action = self.request.get('action')
00073     arg_counter = 0;
00074     args = ()
00075     while True:
00076       arg = self.request.get('arg' + str(arg_counter))
00077       arg_counter += 1
00078       if arg:
00079         args += (simplejson.loads(arg),);
00080       else:
00081         break;
00082     result = getattr(self, action)(*args)
00083     self.response.out.write(simplejson.dumps((result)))

00087   def AddLocation(self, latd, longd, name):
00088     user = users.GetCurrentUser()
00089     query = datastore.Query('User')
00090     query['user ='] = user
00091     if (query.Count() == 0):
00092       user_entity = datastore.Entity('User')
00093       user_entity['user'] = user
00094       datastore.Put(user_entity)
00095     location = datastore.Entity("Location")
00096     location['user'] = user
00097     location['latd'] = latd
00098     location['longd'] = longd
00099     location['name'] = name
00100     datastore.Put(location)
00101     # Add a few additional attributes so that they are available to
00102     # the remote user.
00103     location['user'] = location['user'].email()
00104     location['key'] = str(location.key())
00105     return location
