00034 class GqlEncoder(simplejson.JSONEncoder):
00035   
00036   """Extends JSONEncoder to add support for GQL results and properties.
00037   
00038   Adds support to simplejson JSONEncoders for GQL results and properties by
00039   overriding JSONEncoder's default method.
00040   """
00041   
00042   # TODO Improve coverage for all of App Engine's Property types.

00044   def default(self, obj):
00045     
00046     """Tests the input object, obj, to encode as JSON."""
00047 
00048     if hasattr(obj, '__json__'):
00049       return getattr(obj, '__json__')()
00050 
00051     if isinstance(obj, db.GqlQuery):
00052       return list(obj)
00053 
00054     elif isinstance(obj, db.Model):
00055       properties = obj.properties().items()
00056       output = {}
00057       for field, value in properties:
00058         output[field] = getattr(obj, field)
00059       return output
00060 
00061     elif isinstance(obj, datetime.datetime):
00062       output = {}
00063       fields = ['day', 'hour', 'microsecond', 'minute', 'month', 'second',
00064           'year']
00065       methods = ['ctime', 'isocalendar', 'isoformat', 'isoweekday',
00066           'timetuple']
00067       for field in fields:
00068         output[field] = getattr(obj, field)
00069       for method in methods:
00070         output[method] = getattr(obj, method)()
00071       output['epoch'] = time.mktime(obj.timetuple())
00072       return output
00073 
00074     elif isinstance(obj, time.struct_time):
00075       return list(obj)
00076 
00077     elif isinstance(obj, users.User):
00078       output = {}
00079       methods = ['nickname', 'email', 'auth_domain']
00080       for method in methods:
00081         output[method] = getattr(obj, method)()
00082       return output
00083 
00084     return simplejson.JSONEncoder.default(self, obj)
