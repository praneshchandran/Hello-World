00023 class Lobby(db.Model):
00024   """ Our DB representation of the lobby - basically just a list of players
00025   """
00026   # player is a reference to an individual user in the lobby
00027   player = db.UserProperty(required=True)
00028   
00029   # The timestamp that the user entered the lobby
00030   entered = db.DateTimeProperty(auto_now_add=True)
00031 
00032   # The timestamp updated on every ping from the user's browser. We use this
00033   # to detect when a user has left the lobby (when it's older than about 30
00034   # seconds)
00035   last_update = db.DateTimeProperty(auto_now=True)
00036 
00037 # How long it takes for us to decide a user has left the lobby
00038 EXPIRATION_TIME = datetime.timedelta(seconds=60)
00039 
00040 def in_lobby(user):
00041   """ The user is in the lobby, so update the data store """
00042   get_lobby_for_user(user)
00043 
00044 def get_lobby_for_user(user):
00045   """ Find a non-expired lobby for this user, or create a new one """
00046   entries = get_lobby_entries_for_user(user)
00047   expires = get_expiration_time()
00048   for entry in entries:
00049     if (entry.last_update > expires):
00050       # We've found an entry - update the time and return it
00051       entry.put()
00052       return entry
00053     else:
00054       # Entry is out of date - delete it
00055       entry.delete()
00056 
00057   # Have to create an entry - this has a minor race condition where the
00058   # user could enter the lobby simultaneously from different browsers,
00059   # but it's not worth the trouble to avoid (at worst, people will see
00060   # two entries for the user in the lobby)
00061   result = Lobby(player = user)
00062   result.put()
00063   return result
00064 
00065 def leave_lobby(user):
00066   """ Called when a user has left the lobby - cleans up any lobby entries
00067       (there should never be more than one).
00068   """
00069   entries = get_lobby_entries_for_user(user)
00070   for entry in entries:
00071     entry.delete()
00072 
00073 def get_lobby_entries_for_user(user):
00074   return Lobby.gql("WHERE player = :1", user)
00075 
00076 def get_expiration_time():
00077   return datetime.datetime.now() - EXPIRATION_TIME
00078 
00079 def lobby_list(limit):
00080   """ Gets a list of users in this lobby """
00081   expires = get_expiration_time()
00082   entries = Lobby.gql("WHERE last_update > :time LIMIT %d" % limit, 
00083                       time=expires)
00084   return entries
