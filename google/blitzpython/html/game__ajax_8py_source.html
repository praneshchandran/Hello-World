00232     """ Our handler for HTTP POST requests - this posts a user's move to the
00233         given game. User must be a participant in that game. This attempts to
00234         be idempotent, so if the same move is posted twice in a row, the second
00235         post is ignored.
00236         
00237         URL Path should be one of:
00238           /game_ajax/<id of game to join>/join
00239           (no post params)
00240 
00241           /game_ajax/<id of game to post a move to>/move
00242             POST params are:
00243               move: <string data in ICCF format>
00244                     Special values: 'resign', 'draw' causes a resignation/draw
00245                     Moves that start with '#' denote a checkmate
00246               time: updated time
00247           /game_ajax/<id of game to post a chat to>/chat
00248             POST params are:
00249               chat: <string data>
00250 
00251           /game_ajax/<id of game to update time>/time
00252             POST params are:
00253               time: milliseconds left for user
00254             
00255         Response:
00256           HTTP errors:
00257             http.HTTP_FORBIDDEN if you try to join a game that is already full
00258               or try to move out of turn
00259             http.HTTP_UNAUTHORIZED if you aren't a participant and try to move
00260               or chat
00261             
00262           JSON-encoded response:
00263             none
00264     """
00265     game_to_modify = self._get_game_to_modify(user)
00266     if game_to_modify:
00267       path_list = self.request.path.strip('/').split('/')
00268       command = path_list[2]
00269       if command == 'join':
00270         result = game_to_modify.join(user)
00271         if not result:
00272           # Error - game is already full, or closed
00273           self.error(http.HTTP_FORBIDDEN)
00274       elif command == 'move':
00275         # User is posting a move
00276         move = self.request.get('move')
00277         timer = self.request.get('time')
00278         victor = None
00279         is_resignation = False
00280         if timer:
00281           timer = int(timer)
00282         if move:
00283           # Handle special endgame move commands 
00284           if move == 'resign':
00285             # User resigned
00286             victor = get_player_number(game_to_modify, user, True)
00287             is_resignation = True
00288           elif move == 'draw':
00289             victor = 0
00290           elif move[0] == '#':
00291             # User won by checkmate
00292             victor = get_player_number(game_to_modify, user, False)
00293 
00294           if not game_to_modify.update(user, move, timer, victor):
00295             # User tried to move when it was not his turn
00296             self.error(http.HTTP_FORBIDDEN)
00297           else:
00298             # User moved - let's notify the other player if this is an untimed
00299             # game.
00300             if game_to_modify.game_type == gamemodel.GAME_TYPE_CHESS:
00301               if get_player_number(game_to_modify, user, False) == 1:
00302                 notify_email = game_to_modify.player2.email()
00303               else:
00304                 notify_email = game_to_modify.player1.email()
00305               self.email_status(game_to_modify, notify_email, user.nickname(),
00306                                 victor, is_resignation)
00307       elif command == 'time':
00308         # User is posting a time update
00309         timer = int(self.request.get('time'))
00310         game_to_modify.update(user, timer=timer)
00311       elif command == 'chat':
00312         # User is posting a chat
00313         data = self.request.get('chat')
00314         if data:
00315           chat.add_chat(str(game_to_modify.key()), user, data, GAME_CHAT_LIMIT)

00362 def get_player_number(gameObj, user, opponent):
00363   """ Returns 1 or 2, depending on whether player 1 or player 2 is the user
00364       Param: user - user to operate on
00365              opponent - True to return the *other* player's number, else False 
00366                         to return the user's number
00367   """
00368   if user == gameObj.player1:
00369     player = 1
00370   else:
00371     player = 2
00372 
00373   if opponent:
00374     # Return the other player (3-2=1 and 3-1=2)
00375     return 3 - player
00376   else:
00377     return player
