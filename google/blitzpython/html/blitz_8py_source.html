00029 class ForceLoginPage(webapp.RequestHandler):
00030   """ Base class of all our normal request handlers which makes sure the user
00031       is logged in before rendering the page.
00032   """
00033 
00034   def get(self):
00035     """ Handler for get - redirects to the login page if necessary, otherwise
00036         calls our subclass to generate the response.
00037     """
00038     user = users.GetCurrentUser()
00039     if user:
00040       response = self.Get(user)
00041       if response:
00042         self.response.out.write(response)
00043     else:
00044       self.redirect(users.CreateLoginURL(self.request.uri))

00059 class HistoryPage(ForceLoginPage):
00060   """ Renderer for the history page """
00061   template_path = os.path.join(os.path.dirname(__file__), 'history.html')
00062   def Get(self, user):
00063     completed_games = gamemodel.completed_games_list(user)
00064 
00065     # Build up a string describing the user's record
00066     wins = losses = draws = 0
00067 
00068     for game in completed_games:
00069       if game.is_victory(user):
00070         wins += 1
00071       elif game.is_loss(user):
00072         losses += 1
00073       else:
00074         draws += 1
00075 
00076     record_str = (pluralize(wins, "win, ", "wins, ") + 
00077                   pluralize(losses, "loss, ", "losses, ") + 
00078                   pluralize(draws, "draw", "draws"))
00079 
00080     template_values = {
00081       'user' : user,
00082       'completed_games_list' : completed_games,
00083       'record_string' : record_str,
00084       'logout_url' : users.CreateLogoutURL('/'),
00085       'mainContentClass' : 'historyContent'
00086       }
00087     return template.render(HistoryPage.template_path, template_values)

00089 def pluralize(count, singular, plural):
00090   """ Creates a singular or plural phrase depending on the passed numeric value
00091   """
00092   phrase = singular if count == 1 else plural
00093   return "%d %s" % (count, phrase)
