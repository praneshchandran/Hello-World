00019 class Chat(db.Model):
00020   """ Our DB representation of a chat stream - basically just a long list of
00021       timestamped strings.
00022   """
00023 
00024   # The index of the first chat item in our list
00025   index = db.IntegerProperty(default=1)
00026 
00027   # The name of the chat (used to look up a specific chat, for example the
00028   # lobby chat.
00029   name = db.StringProperty()
00030 
00031   # The list of chat strings the format of each string is:
00032   #   nickname\ndata
00033   data = db.StringListProperty()
00034 
00035   # The maximum number of items we allow in the list (if None, then there's no
00036   # limit)
00037   limit = db.IntegerProperty()
00038 
00039   def get_chat_list(self, index):
00040     """ Gets all chats that are newer than the passed msgid. Since the
00041         clients all poll fairly rapidly, this should never return more than
00042         a few items """
00043     if index <= self.index:
00044       return self.data
00045     else:
00046       return self.data[index-self.index: ]
00047 
00048   def to_dict(self, msgid):
00049     """ Given a msgid, returns all chats that have happened since this
00050         msgid in dict format.
00051         Return: dict object in this format:
00052           'msg_id': 12345,             // The msg_id to pass up on the next
00053                                        // update
00054           'data': [
00055             { 'author': "player 1",    // Omitted if this is a system msg
00056               'data': "blah blah blah"
00057             },
00058             ...more chats...
00059           ]
00060       
00061     """
00062     result = {}
00063     result['msg_id'] = self.index + len(self.data)
00064     chat_result = []
00065     if msgid:
00066       chats = self.get_chat_list(msgid)
00067       for chat_item in chats:
00068         dict = {}
00069         chat_split = chat_item.split("\n")
00070         dict['author'] = chat_split[0]
00071         dict['data'] = chat_split[1]
00072         chat_result.append(dict)
00073     result['data'] = chat_result
00074     return result

00077 def add_chat(name, user, data, limit):
00078   """ Adds a chat entry to this chat stream, using a transaction to ensure
00079       atomicity.
00080   """
00081   chat = get_chat(name, limit)
00082   db.run_in_transaction(txn_add_chat, chat.key(), user, data)

00084 def txn_add_chat(key, user, data):
00085   """ Called within a transaction - adds data to the chat stream
00086   """
00087   chat = Chat.get(key)
00088   chat.data.append(user.nickname() + "\n" + data)
00089   # Toss out the oldest values if we've exceeded our limit
00090   if chat.limit and len(chat.data) > chat.limit:
00091     num_items_to_toss = len(chat.data) - chat.limit
00092     chat.index += num_items_to_toss
00093     chat.data = chat.data[num_items_to_toss:]
00094   chat.put()
00095

00097 def get_chat(name, limit):
00098   """ Gets a chat with the passed name, or creates one with the passed limit
00099       if none exists yet
00100   """
00101   return db.run_in_transaction(txn_get_chat, name, limit)
00102 
00103 def txn_get_chat(name, limit):
00104   """ Implementation of get_chat() that runs in a transaction
00105   """
00106   chat = Chat.get(db.Key.from_path('Chat', name))
00107   if not chat:
00108     # Need to create one
00109     chat = Chat(key_name=name, name=name, limit=limit)
00110     chat.put()
00111   return chat
