00043   def Get(self, user):
00044     """ Sends down a list of players and games for the lobby in json format.
00045 
00046         The URL format is:
00047            GET /lobby/useremail/<opt_msgid>
00048            
00049         Fetches a list of games for the passed user.
00050 
00051         If the opt_msgid is present, we return a list of chats that were sent
00052         after that msgid, otherwise the chat list will be empty
00053 
00054         Response is a Javascript object with 4 attributes:
00055         {
00056           'player_list': ["player 1", "player 2"...],  // string array
00057 
00058           'game_list': [
00059             {
00060              // See Game.to_dict() for information about the format of
00061              // each game
00062             },
00063             ...more games...
00064           ],
00065 
00066           'chat_list': {
00067             'msg_id': 12345,             // The msg_id to pass up on the next
00068                                          // update
00069             data: [
00070               { 'author': "player 1",    // Omitted if this is a system msg
00071                 'data': "blah blah blah"
00072               },
00073               ...more chats...
00074             ]
00075           }
00076         }
00077         
00078     """
00079     self.response.headers['Content-Type'] = 'text/javascript'
00080     # Parse the path into distinct entities
00081     path_list = self._get_path_list(user)
00082     if path_list:
00083       # Get the msgid if any
00084       msgid = 0
00085       if len(path_list) == 3:
00086         try:
00087           msgid = int(path_list[2])
00088         except ValueError:
00089           # Just ignore this and leave msgid as 0
00090           msgid = 0
00091             
00092       # Let the lobby know the user is here
00093       lobby.in_lobby(user)
00094 
00095       # Build our response and write it out
00096       response = {}
00097       response['player_list'] = self.get_player_list()
00098       response['game_list'] = self.get_game_list(user)
00099       response['chat_list'] = chat.get_chat("lobby", 50).to_dict(msgid)
00100       self.response.out.write(simplejson.dumps(response))
00101

00127   def _get_path_list(self, user):
00128     # Parse the path into distinct entities
00129     path_list = self.request.path.strip('/').split('/')
00130     if len(path_list) < 2:
00131       # Invalid query
00132       self.error(http.HTTP_ERROR)
00133       self.response.out.write('Invalid request')
00134     else:
00135       # user email passed - make sure this is that user
00136       email = urllib.unquote(path_list[1])
00137       if (users.IsCurrentUserAdmin() or user.email() == email):
00138         return path_list
00139       else:
00140         self.error(http.HTTP_UNAUTHORIZED)
00141         self.response.out.write('Permissions error for ' + email)
00142     return None

00152   def get_game_list(self, user):
00153     """ Get a list of dict objects corresponding to the active games.
00154         The list includes a max of 50 games that match the following criteria:
00155         1) status is GAME_STATUS_OPEN
00156         2) status is GAME_STATUS_INVITED and the user is one of the players
00157         3) status is GAME_STATUS_ACTIVE and the game is public *or* the user
00158            is one of the players.
00159 
00160         The list is sorted to have the user's games at the top, followed by
00161         open games, followed by public games, each section sorted to have
00162         most recently modified games first
00163     """
00164     # First, get all the user's games
00165     user_games = gamemodel.games_by_user_list(user)
00166     
00167     # Now get list of open and active games,  sorted to have the open games 
00168     # first
00169     active_games = gamemodel.public_game_list()
00170     # Filter to just have the ones that don't involve the user so we don't have 
00171     # duplicates
00172     active_games = filter(lambda gameObj: gameObj.player1 != user and gameObj.player2 != user, active_games)
00173     
00174     # Now render these in dict representation
00175     return map(lambda gameObj: gameObj.to_dict(user), 
00176                list(user_games) + list(active_games))
